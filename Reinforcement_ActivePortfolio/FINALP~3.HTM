# helper functions - not necessarily being used in training

def d1(S, K, sigma, tau):
    return (np.log(S/K) + 1/2 * tau * (sigma)**2) / (sigma * np.sqrt(tau))

def BS(S, K, sigma, tau):
    if tau != 0:
        return S*norm.cdf(d1(S, K, sigma, tau)) - K * norm.cdf(d1(S, K, sigma, tau)) - 1/2 * sigma * np.sqrt(tau)
    else:
        return max(S - K, 0)
    
def sigmoid(x):
    return 1/(1 + np.exp(-x))

def sigmoid_grad(x):
    return np.exp(-x)/(1 + np.exp(-x))**2

def my_log(x):
    if x <= 0.1: 
        return x - 1 # approx log(x) for small x
    else:
        return np.log(x)

class policy:
    '''This class holds methods relevant for the policy function'''
    def __init__(self, possible_actions, theta, K):
        self.min_total = -100 # min total position
        self.max_total = 0 # max total position
        
        self.actions = possible_actions #[-100,100]
        self.K = K
        self.theta = theta
        
    def get_theta(self):
        return self.theta
    
    def set_theta(self, theta):
        self.theta = theta
        
    def get_pref(self, S_t, tau, n, C_t):
        '''calculate perference function'''
        x = ( (self.actions + n)/100 + C_t/S_t )**2 # -x is maximized when new_action + current_position = 100*C_t/S_t
        
        # delta*S + C = 0
        # delta = -C/S
        
        # print statements for debugging
        #print('S_t', S_t)
        #print('C_t', C_t)
        #print('x', x)
        #print('mu', mu)
        #print('sd', sd)
        
        pref = -(1000/np.abs(self.theta))*x # theta acts like a std dev for our pref function
        return pref
    
    def get_prob(self, S_t, tau, n, C_t):
        '''calculate probability function: softmax of preference function'''
        pref = self.get_pref(S_t, tau, n, C_t)
        bounds = (self.actions + n >= self.min_total)*(self.actions + n <= self.max_total) # exclude actions that exceed position limit
        prob = np.exp(pref)*bounds
        return prob/np.sum(prob)
    
    def get_pref_grad(self, S_t, tau, n, C_t):
        '''calculate gradient of preference function'''
        x = ( (self.actions + n)/100 - C_t/S_t )**2
        pref_grad = (1000/np.abs(self.theta))*x*np.sign(self.theta) 
        return pref_grad
        #return pref_grad*(self.actions + n >= self.min_total)*(self.actions + n <= self.max_total)
        
    def get_log_prob_grad(self, S_t, tau, n, C_t):
        '''calculate gradient of log probability function'''
        pref = self.get_pref(S_t, tau, n, C_t)
        pref_grad = self.get_pref_grad(S_t, tau, n, C_t)
        bounds = (self.actions + n >= self.min_total)*(self.actions + n <= self.max_total)
        log_prob_grad = pref_grad - ( 1 / np.sum(np.exp(pref)*bounds) ) * np.sum(np.exp(pref)*pref_grad*bounds)

        return log_prob_grad
        #return #log_prob_grad*(self.actions + n >= self.min_total)*(self.actions + n <= self.max_total)
    
            
    
    
class control:
    def __init__(self, policy, possible_actions, T):
        self.policy = policy
        self.possible_actions = possible_actions
        self.T = T
        
        self.action_seq = np.zeros(T) # action at each time step for one episode
        self.reward_seq = np.zeros(T-1) # reward at each time step for one episode
        self.cum_pos = np.zeros(T) # cumulative position at each time step for one episode
        self.log_prob_grad_seq = np.zeros(T) # log probability gradient at each time step for one episode
        
        ## from wan
        self.state_S = np.zeros(T)
        self.state_Tau = np.zeros(T)
        self.state_n = np.zeros(T)
        self.state_C = np.zeros(T)
        
        
        
    def get_state(self):
        return self.state_S , self.state_Tau,self.state_n,self.state_C
        
          
    def get_action_seq(self):
        return self.action_seq
    
    def get_reward_seq(self):
        return self.reward_seq
    
    def get_cum_pos(self):
        return self.cum_pos
    
    def get_log_prob_grad_seq(self):
        return self.log_prob_grad_seq
    
    def get_policy(self):
        return self.policy
    
    def set_policy(self, policy):
        self.policy = policy
        
    def clear(self):
        '''Reset all arrays'''
        self.action_seq = np.zeros(self.T)
        self.reward_seq = np.zeros(self.T-1)
        self.cum_pos = np.zeros(self.T)
        self.log_prob_grad_seq = np.zeros(self.T)
        
    def perform_control(self, stock_seq, call_seq):
        '''Performs control for on one episode'''
        
        S, C = 0, 0 # stock price, call price
        n, pnl = 0, 0 # cumulative position, cumulative pnl
        
        for t in range(1, self.T+1):
                      
            S_prev, C_prev = S, C # previous prices for pnl calculation
            S, C = stock_seq[t-1], call_seq[t-1] # current prices
            
            action = np.random.choice(self.possible_actions, p=self.policy.get_prob(S, self.T-t, n, C)) # select action according to policy
            self.action_seq[t-1] = action # store action
            
            
            if t > 1:
                pnl += (action+n)*S - n*S_prev + 100*C - 100*C_prev # calculate pnl
                self.reward_seq[t-2] = -(1/self.T)*np.abs(pnl) # store reward    
            
            log_prob_grad = self.policy.get_log_prob_grad(S, self.T-t, n, C) 
            self.log_prob_grad_seq[t-1] = log_prob_grad[action+100] # store log probability gradient
            self.state_S[t-1] = S
            self.state_Tau[t-1] = 30-t
            self.state_n[t-1] = n
            self.state_C[t-1] = C
            
            n += action # update position
            self.cum_pos[t-1] = n # store position

batch_size = 64 # number of episodes to use per iteration
total_updates = 10000 # number of training steps
theta = 50 # initial guess

def alpha(t,initial_learning_rate, T=total_updates):
    '''Learning rate'''
    #initial_learning_rate = 1
    decay_rate = 0.1 # after tuned from problem 1
    return initial_learning_rate * decay_rate ** (t / T)
########################################

def alpha_w(t,initial_learning_rate,decay_rate_w, T=total_updates):
    '''Learning rate baseline'''
    #initial_learning_rate = 0.000001
    return initial_learning_rate * decay_rate_w ** (t / T)

def init_weight(ValueInitialBaseline):
    """Initial weight of linear baseline"""
    v = ValueInitialBaseline
    return np.array([v,v,v,v,v]) # 1 constant  4 features

########################################


def main_function_baseline(initial_learning_rate,initial_learning_rate_w,decay_rate_w
                           , ValueInitialBaseline, batch_size,total_updates,theta):

    startTime = time.time()
    num_training_episodes = train_pd['episode'].nunique()

    possible_actions = np.array(range(-100,101))
    K = 12.5
    T = 30

    train_reward = np.zeros(total_updates)
    cum_train_reward = np.zeros(total_updates)

    my_policy = policy(possible_actions, theta, K)
    my_control = control(my_policy, possible_actions, T)
    
    updateDict = { 'rounds': []
              , 'total_updates': []
              , 'train_rewards': []
              , 'cummulative_train_rewards': []
              , 'thetas': []
              , 'alphaGradients': []
              , 'gradients': []
              , 'Delta_now' : []
            }

    weight_now = init_weight(ValueInitialBaseline)
    V_now = np.sum(weight_now[0] + (weight_now[1]* 10)+ (weight_now[2]*(29)) + (weight_now[3]*-50 ) + (weight_now[4]*0.082495))
    V_now = np.repeat(V_now,repeats=30)
    
    for i in range(total_updates):

        episodes = np.random.randint(1,num_training_episodes+1,batch_size)

        my_policy.set_theta(theta)
        my_control.clear()
        my_control.set_policy(my_policy)

        gradient = 0
        batch_reward = 0

        for j in range(batch_size):

            # get episode
            episode = episodes[j]
            idx = train_pd.loc[train_pd.episode==episode].first_valid_index()
            df = train_pd.iloc[idx:idx+T]
            S = np.array(df['S'])
            C = np.array(df['C'])

            # perform control on episode
            my_control.perform_control(S, C)

            # get info needed for gradient
            reward = np.sum(my_control.get_reward_seq())



            s1, s2 ,s3,s4 = my_control.get_state()
            
            ######## Baseline #########
            V_old = V_now
            V_now = weight_now[0] + (weight_now[1]* s1)+ (weight_now[2]*(s2)) + (weight_now[3]*s3 ) + (weight_now[4]*s4)

            Delta_now = reward - np.sum(V_now)
            #V_diff = V_old - V_now
            V_diff =  -np.abs(np.abs(V_now) - np.abs(np.insert(my_control.get_reward_seq(), 0, 0) ))


            weight_now[0] = weight_now[0]+  (-2*alpha_w(i,initial_learning_rate_w,decay_rate_w)* (1/30) * np.sum(V_diff)   )
            weight_now[1] = weight_now[1]+  (-2*alpha_w(i,initial_learning_rate_w,decay_rate_w)* (1/30) * Delta_now  * V_diff@s1  )   
            weight_now[2] = weight_now[2]+  (-2*alpha_w(i,initial_learning_rate_w,decay_rate_w)* (1/30) * Delta_now  * V_diff@s2  )   
            weight_now[3] = weight_now[3]+  (-2*alpha_w(i,initial_learning_rate_w,decay_rate_w)* (1/30) * Delta_now  * V_diff@s3  )   
            weight_now[4] = weight_now[4]+  (-2*alpha_w(i,initial_learning_rate_w,decay_rate_w)* (1/30) * Delta_now  * V_diff@s4  )   

            log_prob_grad = np.sum(my_control.get_log_prob_grad_seq())

            batch_reward += reward
            gradient += Delta_now*log_prob_grad


        gradient = gradient/batch_size
        theta = theta + alpha(i,initial_learning_rate)*gradient # update theta

        train_reward[i] = batch_reward/batch_size # average reward for iteration

        if i == 0:
            cum_train_reward[i] = batch_reward/batch_size
        else:
            cum_train_reward[i] = (i*cum_train_reward[i-1] + batch_reward/batch_size)/(i+1) # cumulative average reward

        if (i+1) % (total_updates/10) == 0:
            endTime = time.time()
            print("Iter %d/%d - Reward: %.2f  Rolling Avg: %.2f  Theta: %.3f  LR*Gradient: %.3f  Gradient: %.3f"%(i+1, total_updates, train_reward[i], cum_train_reward[i], theta, alpha(i,initial_learning_rate)*gradient, gradient))
            print("Delta_now: %.2f Processing time : %.1f "%(Delta_now,endTime - startTime))
            startTime = time.time()

        updateDict['rounds'].append(i+1)
        updateDict['total_updates'].append(total_updates)
        updateDict['train_rewards'].append(train_reward[i])
        updateDict['cummulative_train_rewards'].append(cum_train_reward[i])
        updateDict['thetas'].append(theta)
        updateDict['alphaGradients'].append(alpha(i,initial_learning_rate)*gradient)
        updateDict['gradients'].append(gradient)
        updateDict['Delta_now'].append(Delta_now)
        
    return pd.DataFrame(updateDict)
